22 function kVIS_dataReplayStart_Callback(hObject, ~)
23 
24 %
25 % Restart playback from pause?
26 %
27 t = timerfind('Tag','HeartbeatTimer');
28 t2 = timerfind('Tag','DataTimer');
29 
30 if ~isempty(t)
31 
32  start(t)
33  start(t2)
34 
35  pauseBtn = findobj('Tag','pausePlaybackToggle');
36  pauseBtn. Value = 0;
37 
38  disp('Restart')
39  return
40 end
41 
42 %
43 % Start from scratch
44 %
45 [fds, name] = kVIS_getCurrentFds(hObject);
46 
47 if isempty(name)
48  errordlg('Nothing loaded...')
49  % reset toggle button
50  hObject.Value = 0;
51  return
52 end
53 
54 %
55 % Create time indicator line
56 %
57 targetPanel = kVIS_dataViewerGetActivePanel();
58 
59 axesHandle = targetPanel.axesHandle;
60 
61 hold(axesHandle, 'on');
62 
63 xlim = kVIS_getDataRange(hObject, 'XLim');
64 ylim = kVIS_getDataRange(hObject, 'YLim');
65 
66 if isnan(ylim)
67  current_lines = kVIS_findValidPlotLines(axesHandle);
68 
69  if ~isempty(current_lines)
70  ylim(1) = min(arrayfun(@(x) x.UserData.yMin, current_lines));
71  ylim(2) = max(arrayfun(@(x) x.UserData.yMax, current_lines));
72  else
73  % nothing plotted, so create empty time indicator plot
74  ylim = [0 1];
75  axesHandle.XLim = xlim;
76  end
77 end
78 
79 lineHandle = line(axesHandle, [xlim(1) xlim(1)], ylim, 'Color', 'r', 'LineWidth', 2.0);
80 
81 
82 %
83 % Open UDP connection
84 %
85 ip = getpref('kVIS_prefs','dataReplayTargetIP');
86 
87 if ~isempty(ip)
88  BSP_dataReplayMex('UDP_Init', ip)
89 else
90  errordlg('Replay IP address invalid...')
91  return
92 end
93 
94 %
95 % heartbeat message @ 1Hz
96 %
97 t = timer('Period', 1, 'TasksToExecute', Inf, ...
98  'ExecutionMode', 'fixedRate', 'Tag', 'HeartbeatTimer');
99 
100 t.TimerFcn = @kVIS_heartbeatTimerFcn;
101 
102 start(t)
103 
104 %
105 % data messages @ 10Hz
106 %
107 updateFrequency = getpref('kVIS_prefs','dataReplaySpeed');
108 
109 t2 = timer('Period', 1/updateFrequency, 'TasksToExecute', Inf, ...
110  'ExecutionMode', 'fixedRate', 'Tag', 'DataTimer');
111 
112 t2.TimerFcn = @kVIS_dataTimerFcn;
113 
114 %
115 % save data in timer user data
116 %
117 Data = BSP_replayDataPrepareFcn(fds);
118 Data.currentStep = 1;
119 Data.lineHandle = lineHandle;
120 Data.updateFrequency = updateFrequency;
121 Data.sampleRate = mean(diff(Data.Time));
122 
123 t2.UserData = Data;
124 
125 start(t2)
126 
127 end
