21 function kVIS_fftUpdate(~, ~)
22 
23 % active time panel
24 sourcePanel = kVIS_dataViewerGetActivePanel();
25 % link target
26 targetPanel = sourcePanel.linkTo;
27 
28 if isempty(targetPanel)
29  errordlg('Cannot find data source. Ensure the corresponding timeplot is selected for manual update...')
30  return
31 end
32 
33 % plot axes
34 ax = targetPanel.axesHandle;
35 
36 % has frequency range been set? New panel?
37 if isempty(targetPanel.fftRange)
38  targetPanel.fftRange = [0.01 10];
39  ax.XLim = [-inf inf];
40  ax.YLim = [-inf inf];
41 end
42 
43 if isempty(targetPanel.fftType)
44  targetPanel.fftType = 'PSD';
45 end
46 
47 lines = findobj(sourcePanel, 'Type', 'Line');
48 
49 % generate psd
50 if ~isempty(lines)
51 
52  targetPanel.fftYLim = ax.YLim;
53 
54  % clear plot
55  cla(ax);
56 
57  % get selected range
58  xRange = kVIS_getDataRange(gcf, 'XLim');
59 
60  % save axes style (lin/log)
61  xScale = ax.XScale;
62  yScale = ax.YScale;
63 
64  ll = size(lines,1);
65 
66  if ll > 1
67  hold(ax, 'on');
68  else
69  hold(ax, 'off');
70  end
71 
72  fmin = targetPanel.fftRange(1);
73  fmax = targetPanel.fftRange(2);
74 
75  for i=1:ll
76  % generate psd - needs to be done one by one to account for
77  % potentially different sample rates...
78  signal = lines(i).YData';
79  timeVec= lines(i).XData';
80  colour = lines(i).Color;
81 
82  % generate psd for selected range
83  locs = find(timeVec >= xRange(1) & timeVec <= xRange(2));
84  signal = signal(locs);
85  timeVec= timeVec(locs);
86 
87  if any(isnan(signal))
88  errordlg('Signal contains NaN. FFT not possible at the moment...')
89  return
90  end
91 
92  w = [fmin:0.01:fmax]*2*pi;
93 
94  if strcmp(targetPanel.fftType, 'PSD')
95 
96  [p, f] = spect(signal-mean(signal), timeVec, w, 10, 0, 0);
97 
98  plot(ax, f, p, 'Color', colour)
99 
100  elseif strcmp(targetPanel.fftType, 'FFT')
101 
102  [Y, f] = DFT(signal-mean(signal), timeVec, w, 10);
103 
104  plot(ax, f, abs(Y), 'Color', colour)
105 
106  end
107 
108  end
109 
110  grid(ax, 'on');
111 
112  ax.UIContextMenu = kVIS_createFFTContextMenu(ax);
113  ax.XScale = xScale;
114  ax.YScale = yScale;
115 
116  ax.YLim = targetPanel.fftYLim;
117  ax.XLim = [fmin, fmax];
118 
119  xlabel(ax, 'Frequency [Hz]');
120 
121  if strcmp(targetPanel.fftType, 'PSD')
122  ylabel(ax, 'Power Spectral Density (PSD)');
123  oldM = findobj(ax.UIContextMenu, 'Text', 'PSD plot');
124  oldM.Checked = 'on';
125  elseif strcmp(targetPanel.fftType, 'FFT')
126  ylabel(ax, 'FFT Magnitude');
127  oldM = findobj(ax.UIContextMenu, 'Text', 'FFT plot');
128  oldM.Checked = 'on';
129  end
130 
131  handles = guidata(gcf);
132  kVIS_setGraphicsStyle(ax, handles.uiTabDataViewer.plotStyles.AxesB);
133  kVIS_axesResizeToContainer(ax);
134 end

204 function [Y, ff] = DFT(y,t,w,navg)
205 %
206 % SPECT Computes power spectral density estimates for measured time series.
207 %
208 % Usage: [P,f,Y,wdw] = spect(y,t,w,navg,ldw,lplot);
209 %
210 % Description:
211 %
212 % Estimates the one-sided auto spectral density
213 % using data windowing in the time domain to reduce
214 % leakage, and averaging in the frequency domain to
215 % reduce random error. The power spectral density
216 % is computed using frequency vector f=w/(2*pi). The
217 % Fourier transform Y is computed on a fine frequency
218 % grid with navg values inserted between each frequency
219 % point specified in the w vector. Input navg is
220 % the number of averages used to generate each value
221 % of P at the frequency points specified in the
222 % w vector. The variance of the power spectral
223 % density estimates = (1/navg)*100 percent
224 % of the spectral density estimates. Inputs w,
225 % navg, and lplot are optional. The defaults
226 % give the same number of auto spectral density
227 % estimates as time domain points, with variance equal to
228 % 10 percent of the estimated auto spectral densities,
229 % with the plot included.
230 %
231 % Input:
232 %
233 % y = matrix of time series column vectors.
234 % t = time vector.
235 % w = frequency vector, rad/sec.
236 % navg = number of averages in the frequency domain.
237 % ldw = data windowing flag:
238 % = 1 for data windowing (default).
239 % = 0 to omit the data windowing.
240 % lplot = plot flag:
241 % = 1 for auto spectral density plot (default).
242 % = 0 to skip the plot.
243 %
244 % Output:
245 %
246 % P = auto spectral density.
247 % f = vector of frequencies corresponding to the elements of P, Hz.
248 % Y = discrete Fourier transform of y on a fine frequency grid.
249 % wdw = time-domain windowing function.
250 %
251 
252 %
253 % Calls:
254 % cvec.m
255 % czts.m
256 %
257 % Author: Eugene A. Morelli
258 %
259 % History:
260 % 28 Oct 2004 - Created and debugged, EAM.
261 % 11 Nov 2004 - Modified inputs, EAM.
262 % 03 Oct 2005 - Corrected frequency vector endpoint treatment, EAM.
263 %
264 % Copyright (C) 2006 Eugene A. Morelli
265 %
266 % This program carries no warranty, not even the implied
267 % warranty of merchantability or fitness for a particular purpose.
268 %
269 % Please email bug reports or suggestions for improvements to:
270 %
271 % e.a.morelli@nasa.gov
272 %
273 
274 [npts,n]=size(y);
275 %
276 % Time vector information used for frequency scaling.
277 %
278 t=cvec(t);
279 npts=length(t);
280 dt=(t(end)-t(1))/length(t);
281 fs=1/dt;
282 %
283 % Default inputs.
284 %
285 if nargin < 4
286  navg=10;
287 end
288 if any(size(navg)~=1)
289  error('Input navg must be a scalar.')
290 end
291 %
292 % Assemble the frequency vector.
293 %
294 if nargin < 3
295  df=1/(npts*dt);
296  f=[0:df:fs/2]';
297  w=2*pi*f;
298 else
299  w=cvec(w);
300  f=w/(2*pi);
301  df=f(2)-f(1);
302 end
303 nf=length(f);
304 f1=max(min(f),0);
305 f2=min(max(f),fs/2);
306 %
307 % Assemble the fine grid frequency vector.
308 %
309 % Make the number of averages odd,
310 % so that the original frequency grid
311 % can be maintained easily.
312 %
313 if mod(navg,2)==0
314  navg=navg+1;
315 end
316 dff=df/navg;
317 ff=[f1:dff:f2]';
318 %
319 % If lower limit of the frequency range
320 % is not zero, extend the fine grid
321 % frequency vector, to get full accuracy
322 % at the endpoint frequency f1.
323 %
324 if f1 > 0
325  ff=[f1-fix(navg/2)*dff:dff:f2]';
326 end
327 %
328 % Similarly for the upper limit of the
329 % frequency range f2.
330 %
331 if f2 < fs/2
332  ff=[min(ff):dff:f2+fix(navg/2)*dff]';
333 end
334 M=length(ff);
335 
336 wdw=ones(npts,1);
337 %
338 % Complex step for the chirp z-transform.
339 %
340 jay=sqrt(-1);
341 W=exp(-jay*2*pi*dff*dt);
342 %
343 % Stay on the unit circle in the z-plane,
344 % and start at the initial frequency min(ff).
345 %
346 A=1*exp(jay*2*pi*min(ff)*dt);
347 %
348 % Compute the chirp z-transform
349 % for the fine frequency grid.
350 %
351 Y=czts(y.*wdw(:,ones(1,n)),M,W,A);
352 
353 end
