21 function kVIS_plotSignal(hObject, signal, signalMeta, hold_mode, axes_handle, plotFcn)
22 %
23 % Add a signal to the DataViewer plot.
24 % The UserData field of every line is filled with metadata that allows to
25 % identify the data source.
26 % Additionally, a context menu is setup to provide editing features.
27 %
28 warning('off', 'MATLAB:handle_graphics:exceptions:SceneNode')
29 
30 handles=guidata(hObject);
31 
32 %
33 % Hold mode
34 %
35 if hold_mode == 1
36  hold(axes_handle, 'on');
37 else
38  hold(axes_handle, 'off');
39 end
40 %
41 % plot line (linear/log/others)
42 %
43 newLine = feval(plotFcn, axes_handle, signalMeta.timeVec, signal);
44 %
45 % save meta data
46 %
47 newLine.DisplayName = kVIS_generateLabels(signalMeta, []);
48 
49 newLine.UserData = struct();
50 newLine.UserData.signalMeta = signalMeta;
51 
52 newLine.UserData.yMin = min(signal);
53 newLine.UserData.yMax = max(signal);
54 
55 % newLine.UserData.current_offset = current_offset;
56 
57 newLine.UIContextMenu = kVIS_createPlotLineContextMenu(newLine);
58 
59 %% adjust axes limits to data range fields
60 
61 % time axis
62 kVIS_dataRangeUpdate_Callback(hObject, [], 'XLim');
63 
64 % vertical axis - add margins to top and bottom
65 current_lines = kVIS_findValidPlotLines(axes_handle);
66 
67 plotMin = min(arrayfun(@(x) x.UserData.yMin, current_lines));
68 plotMax = max(arrayfun(@(x) x.UserData.yMax, current_lines));
69 
70 marginL = abs(0.1*plotMin);
71 if marginL == 0
72  marginL = 0.1;
73 end
74 
75 marginU = abs(0.1*plotMax);
76 if marginU == 0
77  marginU = 0.1;
78 end
79 
80 % if axes_handle == handles.uiTabDataViewer.axesBot
81 % kVIS_setDataRange(hObject, 'YlLim', [plotMin-marginL plotMax+marginU]);
82 % kVIS_dataRangeUpdate_Callback(hObject, [], 'YlLim');
83 % elseif axes_handle == handles.uiTabDataViewer.axesTop
84 % kVIS_setDataRange(hObject, 'YtLim', [plotMin-marginL plotMax+marginU]);
85 % kVIS_dataRangeUpdate_Callback(hObject, [], 'YtLim');
86 % end
87 
88 % keep plot limits if set to value
89 ylim = kVIS_getDataRange(hObject, 'YLim');
90 if ~any(isnan(ylim))
91 % kVIS_setDataRange(hObject, 'YLim', [plotMin plotMax]);
92  kVIS_dataRangeUpdate_Callback(hObject, [], 'YLim');
93 end
94 
95 %% labels
96 xlabel(axes_handle, 'Time [sec]')
97 
98 if size(current_lines) == 1
99  [ labelStr ] = current_lines.DisplayName;
100  ylabel(axes_handle, labelStr, 'Interpreter', 'latex', 'FontSize', 13)
101 else
102  ylabel(axes_handle, []);
103  lg = legend(axes_handle, current_lines, {}, 'Interpreter', 'latex');
104  lg.ItemHitFcn = @legendCallback;
105  kVIS_setGraphicsStyle(lg, handles.uiTabDataViewer.plotStyles.Legend);
106 end
107 
108 
109 % update all axes styles
110 kVIS_axesResizeToContainer(axes_handle);
111 kVIS_setGraphicsStyle(axes_handle, handles.uiTabDataViewer.plotStyles.AxesB);
112 axes_handle.XRuler.Exponent = 0; % no exp in time stamps
113 end

115 function legendCallback(h, e)
116 h;
117 
118 %
119 % Toggle line visibility when clicking on corresponding legend item
120 %
121 if strcmp(e.Peer.Visible, 'on')
122  e.Peer.Visible = 'off';
123 else
124  e.Peer.Visible = 'on';
125 end
126 end
