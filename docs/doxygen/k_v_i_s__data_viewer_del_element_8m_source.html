25 %> @brief Delete a data viewer panel
26 %>
27 %> @param UI handles
28 %> @param Panel to delete

30 function kVIS_dataViewerDelElement(handles, panel)
31 
32 if isempty(panel)
33  panel = kVIS_dataViewerGetActivePanel();
34 end
35 
36 % remove time axes links
37 kVIS_dataViewerLinkTimeAxes(handles, 'off');
38 
39 % delete listener (if exists)
40 delete(panel.plotChangedListener);
41 
42 % if link target, break link
43 if ~isempty(panel.linkFrom)
44  delete(panel.linkFrom.plotChangedListener);
45  panel.linkFrom.BackgroundColor = getpref('kVIS_prefs','uiBackgroundColour');
46  panel.linkFrom.linkTo = [];
47  panel.linkFrom = [];
48 
49 end
50 
51 % if link source, remove link
52 if ~isempty(panel.linkTo)
53  rm(handles, panel.linkTo);
54 end
55 
56 rm(handles, panel);
57 
58 kVIS_dataViewerLinkTimeAxes(handles, 'x');
59 end

62 function rm(handles, panel)
63 
64 % last plot in column -> remove column
65 if size(handles.uiTabDataViewer.Divider(panel.gridLocation(2)).Contents,1) == 1
66  if panel.gridLocation(2) > 1
67  delete(handles.uiTabDataViewer.Divider(panel.gridLocation(2)))
68  else
69  errordlg('Can''t delete last plot in first column.')
70  return
71  end
72 else
73  delete(panel)
74 end
