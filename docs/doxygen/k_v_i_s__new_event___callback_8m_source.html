21 function kVIS_newEvent_Callback(hObject, ~)
22 %
23 % add new event to list
24 %
25 handles = guidata(hObject);
26 
27 eventList = handles.uiTabEvents.eventTable;
28 
29 %
30 % suggest current xlim as event limits
31 %
32 xlim = kVIS_getDataRange(hObject, 'XLim');
33 %
34 % user input
35 %
36 eType = 'Manual';
37 eIn = num2str(xlim(1));
38 eOut = num2str(xlim(2));
39 eDesc = 'New Event';
40 
41 s = inputdlg({'Type','In','Out','Description'}, 'Event edit:', 1, {eType, eIn, eOut, eDesc});
42 
43 if isempty(s)
44  return
45 end
46 
47 
48 [fds, name] = kVIS_getCurrentFds(hObject);
49 
50 %
51 % find place in event list
52 %
53 rowNew = 1;
54 
55 inTimes = cell2mat(eventList.Data(:,2));
56 
57 if ~isempty(inTimes) % any events yet?
58 
59  eList = fds.eventList;
60 
61  rowNew = find(inTimes < str2double(s{2}), 1, 'last');
62 
63  if isempty(rowNew) % first entry - move all down
64 
65  rowNew = 1;
66 
67  eList = [eList(rowNew) eList(rowNew:end)];
68 
69  else
70  rowNew = rowNew + 1;
71 
72  if rowNew > size(eList,2) % last entry - add to end
73 
74  else % somewhere in the middle - move later events down
75  eList = [eList(1:rowNew) eList(rowNew:end)];
76  end
77  end
78 end
79 
80 % fill new entry
81 eList(rowNew).type = s{1};
82 eList(rowNew).start= str2double(s{2});
83 eList(rowNew).end = str2double(s{3});
84 eList(rowNew).description= s{4};
85 eList(rowNew).plotDef='';
86 
87 fds.eventList = eList;
88 
89 % fds = kVIS_eventTimeSeries(hObject, fds);
90 
91 kVIS_updateDataSet(hObject, fds, name)
92 end
