21 function kVIS_updateChannelList_Callback(hObject, ~, nameSelect)
22 
23 handles = guidata(hObject);
24 %
25 % Get file number
26 %
27 if isempty(handles.uiTabData.groupTree.SelectedNodes)
28  set(handles.uiTabData.channelListbox,'Value',0);
29  set(handles.uiTabData.channelListbox,'String',{});
30  return
31 else
32  fileNo = handles.uiTabData.groupTree.SelectedNodes.Value;
33 end
34 
35 kVIS_fdsUpdateSelectionInfo(hObject, []);
36 
37 handles.uiTabData.channelListboxLabel.String = 'Data Channels';
38 
39 [ fds, ~ ] = kVIS_getCurrentFds(hObject);
40 
41 %
42 % Get the variable labels and units.
43 %
44 if nameSelect == 0
45  handles.uiTabData.channelListBoxMenu1.Checked = 'on';
46  handles.uiTabData.channelListBoxMenu2.Checked = 'off';
47  vars = fds.fdata{fds.fdataRows.varNamesDisp, fileNo};
48 
49  units = fds.fdata{fds.fdataRows.varUnits, fileNo};
50  frames = fds.fdata{fds.fdataRows.varFrames, fileNo};
51 
52  %
53  % decode channel name string if appropriate
54  %
55  for k=1:length(units)
56  if contains(vars{k}, 'UNIT') || contains(vars{k}, 'FRAME')
57  [ SignalInfo ] = kVIS_extractInfoFromSignalName( vars{k} );
58  vars{k} = SignalInfo.DispName;
59  units{k} = SignalInfo.Data.Unit;
60  frames{k} = SignalInfo.Data.Frame;
61  else
62  end
63  end
64 
65 else
66  handles.uiTabData.channelListBoxMenu1.Checked = 'off';
67  handles.uiTabData.channelListBoxMenu2.Checked = 'on';
68  vars = fds.fdata{fds.fdataRows.varNames, fileNo};
69  units = fds.fdata{fds.fdataRows.varUnits, fileNo};
70  frames = fds.fdata{fds.fdataRows.varFrames, fileNo};
71 end
72 
73 
74 if ~isempty(vars)
75  vars = cellfun(@(x) sprintf(' %s', x), vars, 'UniformOutput', false);
76 end
77 
78 if isempty(frames)
79  frames = cell(size(units));
80  frames = cellfun(@(x) sprintf(''), frames, 'UniformOutput', false);
81 end
82 %
83 % Mark the channels that have been assigned.
84 %
85 varlist = {};
86 
87 % % Print error message if not enough var name/unit fields available
88 % data_type = evalin('base',['fds.fdata(1,',num2str(ff,'%.d'),');']);
89 % data_type = data_type{1};
90 % expected_fields = n(ff);
91 % loaded_var_names = length(vars);
92 % loaded_var_units = length(units);
93 %
94 % if (expected_fields > loaded_var_names)
95 % clc
96 % fprintf('\n\nIn %s data type - have %d of %d variable name fields. Filling it out so it still works...\n',data_type,loaded_var_names,expected_fields);
97 % warning('Fix up the var names in UAVmainframe_5_0.m');
98 % % Fill out vars to make it long enough
99 % for ii = 1:(expected_fields-loaded_var_names)
100 % vars{end+1,1} = ' FIX ME NAME!';
101 % end
102 % end
103 % if (expected_fields > loaded_var_units)
104 % clc
105 % fprintf('\n\nIn %s data type - have %d of %d variable unit fields Filling it out so it still works...\n',data_type,loaded_var_units,expected_fields);
106 % warning('Fix up the var units in UAVmainframe_5_0.m')
107 % % Fill out units to make it long enough
108 % for ii = 1:(expected_fields-loaded_var_units)
109 % units{end+1,1} = ' FIX ME UNITS!';
110 % end
111 % end
112 %
113 % % Check for too many fields, means running out of data software...
114 % if (expected_fields<max(loaded_var_names,loaded_var_units))
115 % fprintf('\n\n%s data type\n',data_type);
116 % warning('Too many var names/units loaded, probably means you''re running an out-of-date UAVmainframe')
117 % end
118 
119 varlist = cell(fds.fdataAttributes.nChnls(fileNo), 1);
120 
121 for i=1:fds.fdataAttributes.nChnls(fileNo)
122 
123  % mark non-zero channels
124  if any(fds.fdata{fds.fdataRows.data, fileNo}(:,i)) ~= 0
125  vars = mrk_chnl(vars,i);
126  end
127 
128  %
129  % Include SIDPAC channel numbers, and line up the display.
130  %
131  if nameSelect == 0
132 
133  if isempty(char(units(i)))
134  if i < 10
135  varlist(i)=cellstr([num2str(i),' ',char(vars(i))]);
136  elseif i < 100
137  varlist(i)=cellstr([num2str(i),' ',char(vars(i))]);
138  else
139  varlist(i)=cellstr([num2str(i),'',char(vars(i))]);
140  end
141  elseif isempty(char(frames(i)))
142  if i < 10
143  varlist(i)=cellstr([num2str(i),' ',char(vars(i)),' (',char(units(i)),')']);
144  elseif i < 100
145  varlist(i)=cellstr([num2str(i),' ',char(vars(i)),' (',char(units(i)),')']);
146  else
147  varlist(i)=cellstr([num2str(i),'',char(vars(i)),' (',char(units(i)),')']);
148  end
149 
150  else
151  if i < 10
152  varlist(i)=cellstr([num2str(i),' ',char(vars(i)),' (',char(units(i)),') (',char(frames(i)),')']);
153  elseif i < 100
154  varlist(i)=cellstr([num2str(i),' ',char(vars(i)),' (',char(units(i)),') (',char(frames(i)),')']);
155  else
156  varlist(i)=cellstr([num2str(i),'',char(vars(i)),' (',char(units(i)),') (',char(frames(i)),')']);
157  end
158  end
159 
160  else
161  if i < 10
162  varlist(i)=cellstr([num2str(i),' ',char(vars(i))]);
163  elseif i < 100
164  varlist(i)=cellstr([num2str(i),' ',char(vars(i))]);
165  else
166  varlist(i)=cellstr([num2str(i),'',char(vars(i))]);
167  end
168 
169  end
170 end
171 
172 %
173 % Update the listbox.
174 %
175 set(handles.uiTabData.channelListbox,'Value',1);
176 set(handles.uiTabData.channelListbox,'String',varlist);
177 
178 end

180 function mlab = mrk_chnl(lab,n)
181 %
182 % Mark or clear the nth channel.
183 %
184 mlab=lab;
185 if n > 0
186  name=char(lab(n));
187  name(1)='*';
188  mlab(n)=cellstr(name);
189 else
190  n=abs(n);
191  name(1)=' ';
192  mlab(n)=cellstr(name);
193 end
194 return
