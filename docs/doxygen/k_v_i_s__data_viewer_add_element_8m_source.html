21 function kVIS_dataViewerAddElement(hObject, ~, newColumn)
22 
23 handles = guidata(hObject);
24 
25 l = kVIS_dataViewerGetActivePanel();
26 
27 if isempty(l)
28  % first column
29  column = 1;
30 elseif ~isempty(newColumn)
31  % newly added column
32  column = newColumn;
33 else
34  % same column as active panel
35  column = l.gridLocation(2);
36 end
37 
38 %
39 % add new panel as standard timeplot
40 %
41 np = newPanel(handles, column);
42 
43 %
44 % add axes
45 %
46 ax = axes('Parent', np,...
47  'Units','normalized',...
48  'NextPlot','replacechildren',...
49  'ButtonDownFcn', @kVIS_dataViewerSwitchElement);
50 
51 kVIS_axesResizeToContainer(ax)
52 
53 kVIS_setGraphicsStyle(ax, handles.uiTabDataViewer.plotStyles.Axes);
54 
55 %
56 % set panel properties
57 %
58 np.axesHandle = ax;
59 np.gridLocation = [0, column]; % row can change by deleting panel... WIP
60 np.linkPending = false;
61 np.xLim = kVIS_getDataRange(hObject,'XLim');
62 np.yLim = kVIS_getDataRange(hObject,'YLim');
63 
64 %
65 % make panel active
66 %
67 kVIS_dataViewerSwitchElement(np, [])
68 
69 %
70 % link axes time
71 %
72 pp = findobj('Tag','timeplot');
73 if size(pp,1) > 1
74  kVIS_dataRangeUpdate_Callback(gcf, [], 'XLim')
75  kVIS_dataViewerLinkTimeAxes(handles, 'x');
76 end
77 end
