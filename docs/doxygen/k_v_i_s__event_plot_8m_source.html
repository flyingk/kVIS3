33 function kVIS_eventPlot(hObject, ~)
34 
35 handles=guidata(hObject);
36 
37 
38 % if ~isempty(handles.uiTabDataViewer.showEvents)
39 % h = handles.uiTabDataViewer.showEvents;
40 % delete(h)
41 % end
42 
43 %
44 % selected axes
45 %
46 targetPanel = kVIS_dataViewerGetActivePanel();
47 axes_handle = targetPanel.axesHandle;
48 
49 hold(axes_handle, 'on');
50 
51 ylim = kVIS_getDataRange(hObject, 'YLim');
52 if isnan(ylim)
53  current_lines = kVIS_findValidPlotLines(axes_handle);
54 
55  ylim(1) = min(arrayfun(@(x) x.UserData.yMin, current_lines));
56  ylim(2) = max(arrayfun(@(x) x.UserData.yMax, current_lines));
57 end
58 
59 fds = kVIS_getCurrentFds(hObject);
60 
61 eventList = fds.eventList;
62 
63 if isempty(eventList)
64  errordlg('kVIS_dataViewerChannelListAction: event list empty...')
65  return
66 end
67 
68 for j = 1:size(eventList,2)
69 
70  % relate times to samples
71  in = eventList(j).start;
72  out= eventList(j).end;
73 
74  % create plot
75  pg = polyshape([in out out in],[ylim(1) ylim(1) ylim(2) ylim(2)]);
76 
77  pp(j) = plot(axes_handle, pg, ...
78  'EdgeAlpha',0.4,...
79  'FaceAlpha',0.2);
80 
81  % use context menu for labels
82  m = uicontextmenu();
83  uimenu('Parent', m, 'Label', sprintf('%s: %s', eventList(j).type, eventList(j).description), 'Enable', 'off');
84 
85  pp(j).UIContextMenu = m;
86  pp(j).DisplayName = eventList(j).type;
87  % pp(j).Annotation.LegendInformation.IconDisplayStyle = 'off';
88 
89  % Add a label to each of the plots based on the name of the event
90  eventName = [eventList(j).type,' '];
91  text(out,ylim(2),eventName,'Rotation',90,'FontSize',8, ...
92  'VerticalAlignment','bottom','HorizontalAlignment','right');
93 
94 
95 end
96 
97 handles.uiTabDataViewer.showEvents = pp;
98 
99 guidata(hObject, handles);
100 
101 end
