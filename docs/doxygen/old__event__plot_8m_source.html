21 function [ ] = event_plot(figure_handle, fds, Ch, pts, Style)
22 
23 % import package contents
24 cellfun(@import, KSID.import_package());
25 
26 DefaultStyle = struct();
27 DefaultStyle.Figure = struct();
28 DefaultStyle.Figure.Color = [0.3,0.4,0.58];
29 DefaultStyle.Axes = struct();
30 DefaultStyle.Axes.XColor = 'w';
31 DefaultStyle.Axes.YColor = 'w';
32 DefaultStyle.Axes.GridColor = 'k';
33 DefaultStyle.Axes.MinorGridColor = 'k';
34 DefaultStyle.Legend.FontSize = 16;
35 
36 if isempty(Style)
37  Style = DefaultStyle;
38 else
39  Style = KSID.Util.MergeStructs(Style, DefaultStyle);
40 end
41 
42 assert(isgraphics(figure_handle));
43 
44 axes(figure_handle); % Do not use `figure` here (breaks invisible plotting)
45 set_graphics_style(figure_handle, Style.Figure);
46 
47 % Create buttons
48 OriginalUnits = figure_handle.Units;
49 figure_handle.Units = 'normalized';
50 
51 uicontrol( ...
52  figure_handle, ...
53  'Style', 'pushbutton', ...
54  'String', 'Close', ...
55  'Tag', 'close_inp', ...
56  'Units', 'normalized', ...
57  'Position', [0.87,0.01,0.12,0.04], ...
58  'Callback', @(src, event) close(gcbf()) ...
59 );
60 
61 uicontrol( ...
62  figure_handle, ...
63  'Style', 'pushbutton', ...
64  'String', 'Hide legends', ...
65  'Tag', 'legend_hide_button', ...
66  'Units', 'normalized', ...
67  'Position', [0.67, 0.01, 0.12, 0.04], ...
68  'Callback', @hide_all_legends_Callback ...
69 );
70 
71 figure_handle.Units = OriginalUnits;
72 
73 
74 % Check plot definition.
75 % Columns:
76 % 1: Row
77 % 2: Col
78 % 3: AxesLayout
79 % 4: PlotStyle
80 % 5: Color
81 % 6: Group
82 % 7: Channel
83 % 8: ScaleFactor
84 % 9: UnitOverride
85 % 10: LabelOverride
86 assert(iscell(Ch) && size(Ch, 2) >= 9, 'Invalid plot definition');
87 
88 if size(Ch, 2) < 10 % Column 10: LabelOverride
89  Ch = [ Ch, Ch(:, 7) ]; % Copy Channel to LabelOverride (backwards compatibility)
90 end
91 
92 NoOfPlotRows = max(cell2mat(Ch(:,1)));
93 NoOfPlotCols = max(cell2mat(Ch(:,2)));
94 
95 oldpltindex=0;
96 
97 for i = 1:size(Ch, 1)
98 
99  % get time vector
100  tc = fds_get_channel(fds, Ch{i,6}, 'Time', pts);
101 
102  % hotfix...
103  if tc == -1
104  tc = fds.t;
105  end
106 
107  % get channel data
108  [yp(:,i), label(i), unit(i), fieldno(i), channelno(i)] = fds_get_channel(fds, Ch{i,6}, Ch{i,7}, pts);
109  Labels_TeX = fds.fdata{fds.fdataRows.VarLabels_TeX, fieldno(i)};
110  label(i) = Labels_TeX(channelno(i));
111 
112  % apply scale factor
113  yp(:,i) = yp(:,i) * Ch{i,8};
114 
115  % unit override
116  if ~isempty(Ch{i,9})
117  unit(i) = Ch(i,9);
118  end
119 
120  % LabelOverride
121  if ~isempty(Ch{i, 10})
122  label{i} = getfield(ExtractInfoFromSignalName(Ch{i, 10}), 'TeX_Name');
123  end
124 
125  % create plot index
126  pltindex = (Ch{i,1}-1)*NoOfPlotCols+Ch{i,2};
127 
128  if pltindex == oldpltindex
129  k=k+1;
130  else
131  k=1;
132  clear p labelstr mm ma
133  end
134  oldpltindex = pltindex;
135 
136  subplot(NoOfPlotRows, NoOfPlotCols, pltindex)
137 
138  % axes style
139  if Ch{i,3} == 'L'

141  elseif Ch{i,3} == 'R'
142  yyaxis right
143  else
144  % single plot
145  end
146

148  p(k).LineWidth = 2.0;
149 
150  if ~isempty(Ch{i,4})
151  p(k).LineStyle = Ch{i,4};
152  end
153 
154  if ~isempty(Ch{i,5})
155  p(k).Color = Ch{i,5};
156  end
157 
158  labelstr{k} = sprintf('%s %s', label{i}, unit{i});
159 
160  if k > 1
161  legend_handle = legend(labelstr, 'Interpreter', 'latex');
162  set_graphics_style(legend_handle, Style.Legend);
163  ylabel([])
164  else
165  ylabel(labelstr, 'Interpreter', 'latex');
166  end
167 
168  for jj=1:k
169  mm(jj) = min(p(jj).YData);
170  ma(jj) = max(p(jj).YData);
171  end
172 
173  mn = min(mm);
174  mx = max(ma);
175 
176  if mn ~= mx
177  ylim([mn-abs(mn*0.1) mx+abs(mx*0.1)])
178  end
179 
180  xlim([tc(1) tc(end)])
181 
182  grid on
183  xlabel('time')
184  set_graphics_style(gca, Style.Axes);
185 end
186 
187 % find all axes handle of type 'axes' and empty tag
188 all_ha = findobj( figure_handle, 'type', 'axes', 'tag', '' );
189 linkaxes( all_ha, 'x' );
190 
191 end
