21 function [ fds ] = kVIS_fdsUpgrade(fds)
22 % Upgrade FDS to current storage format.
23 
24 currentVersion = 1.0;
25 
26 assert(isstruct(fds), 'FDS must be a struct');
27 
28 if ~isfield(fds, 'fdsVersion')
29  % legacy format
30  fds.fdsVersion = -1;
31 end
32 
33 while fds.fdsVersion <= currentVersion
34  switch fds.fdsVersion
35  case 1.0
36  fds = UpgradeInfoStructs(fds);
37  break;
38  case 0
39  fds = kVIS_fdsUpgrade_V_0_V_1_0(fds);
40  case -1
41  fds = kVIS_fdsUpgradeLegacy(fds);
42  otherwise
43  error('Unsupported FDS version: %s', fds.fdsVersion);
44  end
45 end
46 
47 if fds.fdsVersion >= 1.0
48  kVIS_fdsValidate(fds);
49 end
50 
51 end
