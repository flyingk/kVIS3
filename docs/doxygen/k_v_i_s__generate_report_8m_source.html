38 % select destination folder

40 outFolder = uigetdir();
41 
42 if outFolder == 0
43  kVIS_terminalMsg('')
44  return
45 end

49 
50 outFile = fullfile(outFolder,'kVIS_report_dev.tex');
51 
52 fidOUT = fopen(outFile,'w');
53 
54 
55 while ~feof(fidIN)
56 
57  l = strip(fgetl(fidIN));
58 
59  if contains(l, '%_kVIS_make_title')
60  kVIS_writeTitle(hObject, fidOUT);
61 
62  elseif contains(l, '%_kVIS_aircraft_properties')
63  kVIS_writeAcDataTable(hObject, fidOUT);
64 
65  elseif contains(l, '%_kVIS_test_info')
66  kVIS_writeTestInfoTable(hObject, fidOUT);
67 
68  elseif contains(l, '%_kVIS_FPRresults')
69  kVIS_writeFPRtable(hObject, fidOUT);
70 
71  elseif contains(l, '%_kVIS_bsp_fcn_eval')
72 
73  fcnName = strsplit(l,{'{','}','[',']'});
74 
75  fileNames = kVIS_writeFcnPlotElement(hObject, fcnName, outFolder);
76 
77  % generate tex with plot names
78  for I = 1:size(fileNames,2)
79  if ~isempty(fileNames{1,I})
80  kVIS_writePlotElement(fidOUT, fileNames(:,I));
81  end
82  end
83 
84  elseif contains(l, '%_kVIS_plot')
85 
86  % generate plots
87  if contains(l, '[')
88  pltName = strsplit(l,{'{','}','[',']'});
89

94  else
95  pltName = strsplit(l,{'{','}'});
96 
97  kVIS_terminalMsg(['Reports: Generating plot ' pltName{2}])
98

101  end
102 
103  % generate tex with plot names
104  for I = 1:size(fileNames,2)
105  if ~isempty(fileNames{1,I})
106  kVIS_writePlotElement(fidOUT, fileNames(:,I));
107  end
108  end
109 
110  %%%% Lorenzo workaround
111  elseif contains(l, '%_kVIS_map_plot')
112 
113  pltCh = strsplit(l,{'{','}'});
114 
115  kVIS_terminalMsg(['Reports: Generating map plot ' pltCh{2}])
116 
117  fileNames = kVIS_writeMapPlotElement(hObject, pltCh{2}, outFolder);
118 
119  % generate tex with plot names
120 
121  if ~isempty(fileNames{1})
122  kVIS_writePlotElement(fidOUT, fileNames);
123  end
124  else
125  fprintf(fidOUT,'%s\n',l);

127 
128 end
129 
130 fclose(fidIN);

133 kVIS_terminalMsg('Report complete.')

149 function [ OutFile ] = TeX_compile(Document)
150  OldPWD = pwd();
151  OldPath = getenv('PATH');
152  try
153  cd(fileparts(Document));
154  if isunix()
155  % Ensure that TeX is on the PATH:
156  setenv('PATH', strjoin({OldPath, '/Library/TeX/texbin/'}, ':'));
157  end
158  setenv('TEX_COMPILE_DOCUMENT', Document);
159  % Run LaTeX four times in order to update all references
160  % (BasicTeX does not include `latexmk`).
161  for I = 1 : 2
162  if isunix()
163  system('pdflatex -shell-escape -interaction=batchmode "%TEX_COMPILE_DOCUMENT%" </dev/null >/dev/null');
164  else
165  system('pdflatex -shell-escape -interaction=batchmode "%TEX_COMPILE_DOCUMENT%"');
166  end
167  end
168  catch
169  end
170  OutFile = regexprep(Document, '(\.tex)?$', '.pdf');
171  if ~exist(OutFile, 'file')
172  error('Failed to compile document %s', Document);
173  end
174  cd(OldPWD);
175  setenv('PATH', OldPath);
176 end
