33 function kVIS_fdsImportDataGroup(hObject, ~)
34 %
35 % Import data from csv file and add to current fds tree
36 %
37 
38 [fds, name] = kVIS_getCurrentFds(hObject);
39 
40 if ~isstruct(fds)
41  errordlg('No data loaded to be assigned.')
42  return
43 end
44 
45 
46 
47 [file, pathname] = uigetfile('*.csv');
48 
49 if file==0
50  return
51 end
52 
53 InputFile = fullfile(pathname,file);
54 
55 Tmp = fopen(InputFile);
56 ExportNames = strsplit(fgetl(Tmp), ',');
57 fclose(Tmp);
58 
59 
60 
61 
62 T = readtable(InputFile);
63 
64 T.Properties;
65 
66 Tr = table2array(T);
67 
68 if iscell(Tr)
69  errordlg('Faulty FTI file - contains non-numerical data. Abort')
70  fds = -1;
71  return
72 end
73 
74 
75 % add data to fds
76 fdata = Tr;
77 
78 varNames = ExportNames';
79 
80 varUnits = cell(size(varNames));
81 varUnits = cellfun(@(x) '', varUnits, 'UniformOutput', false);
82 
83 varFrames = cell(size(varUnits));
84 varFrames = cellfun(@(x) '', varFrames, 'UniformOutput', false);
85 
86 % select location in data tree
87 answ = inputdlg({'New Tree branch','New Group name'},'Select import names',1,{'Imported','FTI Data'});
88 
89 if isempty(answ)
90  return
91 end
92 
93 % add data to tree
94 fds = kVIS_fdsAddDataGroup(fds, answ{1}, answ{2}, varNames, varUnits, varFrames, fdata);
95 
96 kVIS_updateDataSet(hObject, fds, name)
97 end
