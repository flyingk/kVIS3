149 function kVIS_eventEditToggle_Callback(hObject, ~, reset)
150 
151 handles = guidata(hObject);
152 
153 if reset == 1
154  hObject.Value = 0;
155 end
156 
157 if hObject.Value == 1
158  hObject.CData = imread('icons8-edit-36_p.png')-20;
159  handles.uiTabEvents.eventEditToggle = 1;
160 elseif hObject.Value == 0
161  hObject.CData = imread('icons8-edit-36.png');
162  handles.uiTabEvents.eventEditToggle = 0;
163 end
164 
165 guidata(hObject, handles);
