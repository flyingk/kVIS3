21 function kVIS_groupTreeUpdate(hObject, fds, mode)
22 % Update the file tree display, e.g. when selecting a flight from the
23 % flight list.
24 
25 handles=guidata(hObject);
26 
27 tree = handles.uiTabData.groupTree;
28 
29 tree.RootVisible = false;
30 
31 % Clear the tree.
32 delete(tree.Root.Children);
33 
34 % Stop if no dataset is selected (e.g. when the last dataset is closed).
35 if isempty(fds)
36  return;
37 end
38 
39 fdata = fds.fdata;
40 % Create a list of TreeNode objects associated with the fdata entries.
41 fdata_nodes = uiw.widget.TreeNode.empty();
42 
43 % Add nodes for all fdata entries.
44 for idx = 1 : size(fdata, 2)
45 
46  node_name = fdata{fds.fdataRows.groupLabel, idx};
47 
48  % Find the parent node (data group/subsystem/...); if no parent is
49  % specified (parent_idx <= 0), then add the new node at the highest
50  % level in the tree.
51  parent_idx = fdata{fds.fdataRows.treeParent, idx};
52  if parent_idx > 0
53  parent_node = fdata_nodes(parent_idx);
54  else
55  parent_node = tree.Root;
56  end
57 
58  % Create a tooltip indicating the number of variables that is available
59  % at the new node.
60  n_vars = fds.fdataAttributes.nChnls(idx);
61  smplRate = fds.fdataAttributes.sampleRates(idx);
62  switch n_vars
63  case 0
64  tooltip = sprintf('%d Channels', n_vars);
65  case 1
66  tooltip = sprintf('1 Channel - Sample rate: %.1f Hz', smplRate);
67  otherwise
68  tooltip = sprintf('%d Channels - Sample rate: %.1f Hz', n_vars, smplRate);
69  end
70 
71  % Create the node.
72  node = uiw.widget.TreeNode( ...
73  'Name', node_name, ...
74  'Value', idx, ...
75  'TooltipString', tooltip, ...
76  'Parent', parent_node ...
77  );
78 
79  fdata_nodes = [ fdata_nodes, node ]; %#ok<AGROW>
80 
81  if fdata{fds.fdataRows.treeGroupExpanded}
82  node.expand();
83  end
84 
85 end
86 
87 % Restore selection state from information recorded in fds.
88 tree.SelectedNodes = fdata_nodes(cell2mat(fdata(fds.fdataRows.treeGroupSelected, :)));
89 if isempty(tree.SelectedNodes)
90  % Select the first node by default
91  tree.SelectedNodes = fdata_nodes(1);
92 end
93 
94 if strcmp(mode,'search') == false
95  kVIS_updateChannelList_Callback(hObject, [], 0);
96 end
97 
98 end
