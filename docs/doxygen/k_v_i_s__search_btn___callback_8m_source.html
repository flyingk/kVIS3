21 function kVIS_searchBtn_Callback(hObject, ~)
22 
23 handles = guidata(hObject);
24 
25 fds = kVIS_getCurrentFds(hObject);
26 
27 % disp('Search Results:')
28 % disp('===============')
29 % disp(' ')
30 
31 searchStr = handles.uiTabData.groupSearchString.String;
32 
33 % % search for groups
34 % findGroups = contains(fds.fdata(fds.fdataRows.groupLabel, :), searchStr, 'IgnoreCase',true);
35 %
36 % if ~any(findGroups)
37 % disp('No match in group names found...')
38 % disp(' ')
39 % else
40 % GroupResults = fds.fdata{fds.fdataRows.groupLabel, findGroups}
41 % disp(' ')
42 % end
43 
44 % search for channels
45 findChan = {};
46 searchStr_split = split (searchStr, ' ');
47 
48 %% Search engine
49 % - Each word separated by a space is separated query
50 % - If the word starts with a "-" then channels that contains that word are
51 % removed from the filter
52 % - Show channels that fulfils all queries (AND of all queries)
53 % - If no word is filled in the search box, shows all available channels.
54 % - The search considers the name of the group and channel.
55 for i = 1:size(fds.fdata,2)
56  if ~isempty(fds.fdata{fds.fdataRows.varNames, i})
57  % Check all filter criterias
58  a = ones(size(fds.fdata{fds.fdataRows.varNames, i},1), size(fds.fdata{fds.fdataRows.varNames, i},2));
59  for iFilt = 1:length(searchStr_split)
60  if isempty(searchStr_split{iFilt})
61  continue;
62  elseif searchStr_split{iFilt}(1) == '-'
63  a = (a & ~(...
64  contains(lower(fds.fdata{fds.fdataRows.varNames, i}), lower(searchStr_split{iFilt}(2:end))) | ...
65  contains(lower(fds.fdata{fds.fdataRows.groupLabel, i}), lower(searchStr_split{iFilt}(2:end))) ) ...
66  );
67  else
68  a = (a & (...
69  contains(lower(fds.fdata{fds.fdataRows.varNames, i}), lower(searchStr_split{iFilt})) | ...
70  contains(lower(fds.fdata{fds.fdataRows.groupLabel, i}), lower(searchStr_split{iFilt})) ) ...
71  );
72  end
73  end
74  % Consolidate filter results
75  if any(a)
76  channel = find(a==true);
77  for j = 1:length(channel)
78  res = [num2str(i) '/' ...
79  num2str(channel(j)) '/' ...
80  fds.fdata{fds.fdataRows.groupLabel, i} '/' ...
81  fds.fdata{fds.fdataRows.varNames, i}{channel(j)}];
82  findChan = [findChan res];
83  end
84  end
85  end
86 end
87 
88 %% Display results
89 % Reuse Data Channels box to display filter results.
90 if ~isempty(findChan)
91  set(handles.uiTabData.channelListbox,'Value',1);
92  set(handles.uiTabData.channelListbox,'String',findChan);
93  handles.uiTabData.channelListboxLabel.String = 'Search Results';
94 else
95  warndlg('No match for search query found...')
96 end
97 
98 return
99 
100 if ~isempty(findChan)
101 % disp('ChannelResults:')
102 % disp(' ')
103 % for i = 1: length(findChan)
104 % fprintf('%d) %s\n', i, findChan{i})
105 % end
106 
107  sel = listdlg('ListString', findChan, 'ListSize', [300,500],...
108  'SelectionMode','single','PromptString','Search results:','OkString','Select');
109 else
110  sel = [];
111  disp('No match in channel names found...')
112 end
113 
114 %
115 % open tree and plot signal for selected result
116 %
117 if ~isempty(sel)
118 
119  str = strsplit(findChan{sel},'/');
120 
121  % open tree group
122  kVIS_fdsUpdateSelectionInfo(hObject, str2double(str{1}))
123 
124  fds = kVIS_getCurrentFds(hObject);
125 
126  kVIS_groupTreeUpdate(hObject, fds, 'default')
127 
128  % set channel list entry
129  set(handles.uiTabData.channelListbox,'Value', str2double(str{2}));
130 
131  kVIS_channelList_Callback(hObject, [])
132 end
133 end
