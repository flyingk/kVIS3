21 function kVIS_eventExport_Callback(hObject,~)
22 %
23 % export event(s) as individual fds files
24 %
25 
26 % Get the current fild file
27 [fds, name] = kVIS_getCurrentFds(hObject);
28 
29 if isempty(name)
30  errordlg('Nothing loaded...')
31  return
32 end
33 
34 handles = guidata(hObject);
35 
36 eventList = handles.uiTabEvents.eventTable;
37 
38 %
39 % get selected row(s) from user data field, if any
40 %
41 try
42  EventSelected = eventList.UserData.rowSelected;
43 catch
44  EventSelected = [];
45 end
46 
47 if isempty(EventSelected)
48 
49  button = questdlg('Export all events?','Event export','All','Cancel','All');
50 
51 else
52 
53  button = questdlg('Export all or selected events?','Event export','All','Selected','Cancel','All');
54 
55 end
56 
57 if strcmp(button, 'Cancel')
58  return
59 end
60 
61 
62 % promt for file name template
63 prompt = {'Enter file name template:'};
64 dlg_title = 'Export file name';
65 num_lines = 1;
66 def = {[name,'_event']};
67 answer = inputdlg(prompt,dlg_title,num_lines,def);
68 name_template = answer{1};
69 
70 % Get number of events
71 n_events = numel(fds.eventList);
72 
73 %
74 % Export only selected events
75 %
76 if strcmp(button, 'Selected')
77 
78  n_events = EventSelected';
79 else
80 
81  n_events = 1:n_events;
82 end
83 
84 % Loop through each of the events
85 for ii = n_events
86  %
87  % Get time limits
88  %
89  start = fds.eventList(ii).start;
90  stop = fds.eventList(ii).end;
91  eventID = fds.eventList(ii).type;
92 
93  fprintf('Exporting event %3d : %6.1f s to %6.1fs\n',ii,start,stop);
94 
95  %
96  % edit the file
97  %
98  [fdsExport, ~] = kVIS_fdsTrimToTimeRange(fds, [start,stop]); %#ok<ASGLU>
99 
100  %
101  % Save file
102  %
103  fname = [strip(name_template), '_', num2str(ii), '_', strip(eventID)];
104 
105  save(fname, 'fdsExport')
106 
107 end
108 
109 kVIS_terminalMsg('Export Complete.');
110 end
