21 function kVIS_eventListCellSelect_Callback(hObject, eventdata)
22 %
23 % focus on selected event OR edit event fields
24 %
25 
26 %
27 % do not react to weird events
28 %
29 if isempty(eventdata.Indices)
30  return
31 end
32 
33 handles = guidata(hObject);
34 %
35 % check event edit toggle state
36 %

38 
39  eventFocus(hObject, eventdata);
40 
41 else
42 
43  eventEdit(hObject, eventdata);
44 
45 end
46

50 guidata(hObject, handles)
51 end

53 function eventFocus(hObject, eventdata)
54 %
55 % focus on selected event
56 %

58

60 
61 EventID = eventdata.Indices(1);
62 EventCol= eventdata.Indices(2);
63 
64 in = eventList.Data{EventID,2};
65 out = eventList.Data{EventID,3};
66 
67 kVIS_setDataRange(hObject, 'XLim', [in out]);
68 kVIS_dataRangeUpdate_Callback(hObject, [], 'XLim')
69 
70 %

72 %

77 end

79 function eventEdit(hObject, eventdata)
80 %
81 % edit selected event data
82 %
83 handles = guidata(hObject);
84 
85 eventList = handles.uiTabEvents.eventTable;
86 
87 EventID = eventdata.Indices(1);
88 EventCol= eventdata.Indices(2);
89 
90 
91 % edit event fields
92 if EventCol < 5 % string entries
93 
94  eType = eventList.Data{EventID, 1};
95  eIn = num2str(eventList.Data{EventID, 2});
96  eOut = num2str(eventList.Data{EventID, 3});
97  eDesc = eventList.Data{EventID, 4};
98 
99  s = inputdlg({'Type','In','Out','Description'}, 'Event edit:', 1, {eType, eIn, eOut, eDesc});
100 
101  if isempty(s)
102  return
103  end
104 
105  [fds, name] = kVIS_getCurrentFds(hObject);
106 
107  eList = fds.eventList(EventID);
108 
109  eList.type = s{1};
110  eList.start= str2double(s{2});
111  eList.end = str2double(s{3});
112  eList.description = s{4};
113 
114  fds.eventList(EventID) = eList;
115 
116  kVIS_updateDataSet(hObject, fds, name)
117 
118 elseif EventCol == 5 % plot list
119 
120  % get list of loaded custom plot definitions
121  BSP_Name = fieldnames(handles.uiTabPlots.CustomPlots);
122 
123  custom_plot_list = fieldnames(handles.uiTabPlots.CustomPlots.(BSP_Name{1}));
124 
125  [s, ok] = listdlg('PromptString', 'Select a Plot Definition for Event',...
126  'ListString', custom_plot_list, 'SelectionMode', 'single');
127

130  end
131 
132  [fds, name] = kVIS_getCurrentFds(hObject);
133

142 end
