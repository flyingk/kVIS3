21 function [ ] = kVIS_generateCustomPlotM(figure_handle, fds, Ch, pts, Style)
22 
23 DefaultStyle = struct();
24 DefaultStyle.Figure = struct();
25 DefaultStyle.Figure.Color = [0.3,0.4,0.58];
26 DefaultStyle.Axes = struct();
27 DefaultStyle.Axes.XColor = 'w';
28 DefaultStyle.Axes.YColor = 'w';
29 DefaultStyle.Axes.GridColor = 'k';
30 DefaultStyle.Axes.MinorGridColor = 'k';
31 DefaultStyle.Legend.FontSize = 12;
32 DefaultStyle.Legend.Location = 'northeast';
33 
34 if isempty(Style)
35  Style = DefaultStyle;
36 else
37 % Style = KSID.Util.MergeStructs(Style, DefaultStyle);
38 end
39 
40 kVIS_setGraphicsStyle(figure_handle, Style.Figure);
41 
42 main_div = uix.VBox('Parent', figure_handle);
43 
44 plts = uix.Grid('Parent', main_div);
45 plts.Tag = 'plts';
46 
47 ctrls= uix.HButtonBox('Parent', main_div, ...
48  'Backgroundcolor', [0.3,0.4,0.58], ...
49  'Spacing', 50, ...
50  'HorizontalAlignment', 'right');
51 
52 main_div.Heights = [-1 40];
53 
54 % Create buttons
55 uicontrol( ...
56  ctrls, ...
57  'Style', 'pushbutton', ...
58  'String', 'Export .jpg', ...
59  'Callback', @kVIS_reportMode_Callback ...
60 );
61 
62 uicontrol( ...
63  ctrls, ...
64  'Style', 'pushbutton', ...
65  'String', 'Hide legends', ...
66  'Callback', @hide_all_legends_Callback ...
67 );
68 
69 uicontrol( ...
70  ctrls, ...
71  'Style', 'pushbutton', ...
72  'String', 'Close', ...
73  'Callback', @(src, event) close(gcbf()) ...
74 );
75 
76 ctrls.ButtonSize = [100 30];
77 
78 % Check plot definition.
79 % Columns:
80 % 1: Row
81 % 2: Col
82 % 3: AxesLayout
83 % 4: PlotStyle
84 % 5: Color
85 % 6: Group
86 % 7: Channel
87 % 8: ScaleFactor
88 % 9: UnitOverride
89 % 10: LabelOverride
90 assert(iscell(Ch) && size(Ch, 2) >= 9, 'Invalid plot definition');
91 
92 if size(Ch, 2) < 10 % Column 10: LabelOverride
93  Ch = [ Ch, Ch(:, 7) ]; % Copy Channel to LabelOverride (backwards compatibility)
94 end
95 
96 NoOfPlotRows = max(cell2mat(Ch(:,1)));
97 NoOfPlotCols = max(cell2mat(Ch(:,2)));
98 
99 oldpltindex=0;
100 
101 for i = 1:size(Ch, 1)
102 
103  % get time vector
104  tc = kVIS_fdsGetChannel(fds, Ch{i,6}, 'Time');
105 
106  % hotfix...
107  if tc == -1
108  disp('invalid plot def for this dataset...')
109  continue
110  end
111 
112  % get channel data
113  [yp(:,i), meta] = kVIS_fdsGetChannel(fds, Ch{i,6}, Ch{i,7});
114 
115  if ~isstruct(meta)
116  disp('invalid plot def for this dataset...')
117  continue
118  end
119 
120  label = getfield(ExtractInfoFromSignalName(meta.name), 'TeX_Name');
121 
122  unitstr = getfield(ExtractInfoFromSignalName(meta.name), 'TeX_Unit');
123  if isempty(unitstr)
124  unit = unitstr;
125  else
126  unit = ['[' unitstr ']'];
127  end
128 
129  % apply scale factor
130  yp(:,i) = yp(:,i) * Ch{i,8};
131 
132  % unit override
133  if ~isempty(Ch{i,9})
134  unit = ['[' Ch{i,9} ']'];
135  end
136 
137  % LabelOverride
138  if ~isempty(Ch{i, 10})
139  label = getfield(ExtractInfoFromSignalName(Ch{i, 10}), 'TeX_Name');
140  end
141 
142  % create plot index
143  pltindex = (Ch{i,1}-1)*NoOfPlotCols+Ch{i,2};
144 
145  if pltindex == oldpltindex
146  k=k+1;
147  else
148  k=1;
149  clear p labelstr mm ma
150 
151  hh(pltindex) = uipanel('Parent', plts, 'Backgroundcolor', [0.3,0.4,0.58]);
152  ax = axes(hh(pltindex));
153  end
154  oldpltindex = pltindex;
155 
156  % axes style
157  if Ch{i,3} == 'L'
158  yyaxis left
159  elseif Ch{i,3} == 'R'
160  yyaxis right
161  else
162  % single plot
163  end
164 
165  p(k) = plot(ax, tc, yp(:,i)); hold on
166  p(k).LineWidth = 2.0;
167 
168  if ~isempty(Ch{i,4})
169  p(k).LineStyle = Ch{i,4};
170  end
171 
172  if ~isempty(Ch{i,5})
173  p(k).Color = Ch{i,5};
174  end
175 
176  labelstr{k} = sprintf('%s %s', label, unit);
177 
178  if k > 1
179  legend_handle = legend(labelstr, 'Interpreter', 'latex');
180  kVIS_setGraphicsStyle(legend_handle, Style.Legend);
181  ylabel([])
182  else
183  ylabel(labelstr, 'Interpreter', 'latex');
184  end
185 
186  for jj=1:k
187  mm(jj) = min(p(jj).YData);
188  ma(jj) = max(p(jj).YData);
189  end
190 
191  mn = min(mm);
192  mx = max(ma);
193 
194  if mn ~= mx
195  ylim([mn-abs(mn*0.1) mx+abs(mx*0.1)])
196  end
197 
198  xlim([tc(1) tc(end)])
199 
200  grid on
201  xlabel('time')
202  kVIS_setGraphicsStyle(gca, Style.Axes);
203 end
204 
205 plts.Widths = ones(1, NoOfPlotCols)*-1;
206 plts.Heights = ones(1, NoOfPlotRows)*-1;
207 
208 % find all axes handle of type 'axes' and empty tag
209 all_ha = findobj( figure_handle, 'type', 'axes', 'tag', '' );
210 linkaxes( all_ha, 'x' );
211 
212 end
