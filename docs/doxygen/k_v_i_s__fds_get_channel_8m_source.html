21 function [signal, signalMeta] = kVIS_fdsGetChannel(fds, group, channel)
22 
23 % returns the requested channel from the fds structure
24 %
25 % [data, fieldno, channelno] = fds_get_channel(fds, field, channel)
26 %
27 % Matt and Kai 2017
28 
29 % find the array number
30 group_found = 0;
31 groupNo = 1;
32 
33 while (~group_found)
34  % Get field name
35  current_field = fds.fdata{fds.fdataRows.groupLabel, groupNo};
36 
37  % Check for match (case insensitive)
38  if (strcmpi(current_field, group))
39  % Channel has been found
40  group_found = 1;
41 
42  else
43  % Increment counter and try again
44  groupNo = groupNo + 1;
45 
46  % Check for end of array
47  if groupNo > size(fds.fdata,2)
48  signal = -1;
49  signalMeta = -1;
50  fprintf('Can''t find group << %s >>\n', group);
51  return
52  end
53 
54  end
55 
56 end
57 
58 % find the channel number
59 channel_found = 0;
60 channel_idx=1;
61 
62 while (~channel_found)
63  % Get channel name
64  current_channel = fds.fdata{fds.fdataRows.varNames, groupNo}{channel_idx};
65 
66  % Check for match (case sensitive)
67  if (strcmp(strip(current_channel), strip(channel)))
68  % Channel has been found
69  channel_found = 1;
70 
71  else
72  % Increment counter and try again
73  channel_idx = channel_idx + 1;
74 
75  % Check for end of array
76  if channel_idx > length(fds.fdata{fds.fdataRows.varNames,groupNo})
77  signal = -1;
78  signalMeta = -1;
79  fprintf('Can''t find channel << %s >> in group << %s >>\n', channel, group);
80  return
81  end
82 
83  end
84 end
85 %
86 % signal data
87 %
88 signal = fds.fdata{fds.fdataRows.data,groupNo}(:,channel_idx);
89 %
90 % signal meta data as strings, vector
91 %
92 signalMeta.name = fds.fdata{fds.fdataRows.varNames, groupNo}{channel_idx};
93 signalMeta.unit = fds.fdata{fds.fdataRows.varUnits, groupNo}{channel_idx};
94 signalMeta.frame = fds.fdata{fds.fdataRows.varFrames, groupNo}{channel_idx};
95 signalMeta.dispName = fds.fdata{fds.fdataRows.varNamesDisp, groupNo}{channel_idx};
96 signalMeta.dataSet = [];
97 signalMeta.dataGroup = fds.fdata{fds.fdataRows.groupLabel, groupNo};
98 signalMeta.timeVec = fds.fdata{fds.fdataRows.data, groupNo}(:, 1);
99 end
