30 function kVIS_dataViewerExportAllPlots(hObject, ~)
31 
32 kvis = findobj('Tag', 'appWindow');
33 
34 % KVIS uses the GUI Layout toolbox, a set of tools that organizes the
35 % different GUI elements in boxes where more boxes/elements can be added.
36 % We need to look into those elements (see the /+uix package) and find
37 % where are our axes located
38 
39 % 1. Our data to be extracted are the axes and their plots. If we look at
40 % kVIS, all axes are located inside a tab called 'Data Viewer'. First step,
41 % find the data viewer tab
42 kchild = kvis.Children; % Get the kVIS children
43 uixidx = arrayfun(@(x) isa(x, 'uix.VBox'), kchild); % Find the uix element (normally idx 7)
44 uixbox = kchild(uixidx); % Get the uix element
45 flxidx = arrayfun(@(x) isa(x, 'uix.HBoxFlex'), uixbox.Children); % The data viewer is resizable, so we need to find a 'BoxFlex' component
46 flxbox = uixbox.Children(flxidx); % Get the resizable box element
47 tabidx = arrayfun(@(x) isa(x, 'matlab.ui.container.TabGroup'), flxbox.Children);
48 tabgrp = flxbox.Children(tabidx); % Get the tabgroup element
49 tabs = tabgrp.Children; % FInd the datat viewer between the tabs
50 tabttl = {tabs.Title};
51 dataViewerIdx = strcmp(tabttl, 'Data Viewer');
52 dataViewer = tabs(dataViewerIdx);
53 
54 % 2. The data viewer has a resizable horizontal box container that serves
55 % as basis.
56 dvwrbox = dataViewer.Children(1);
57 
58 % 3. Then, every added column creates a resizable vertical box container.
59 % Note that columns are added in the reverse order (from right to left)
60 columns = dvwrbox.Children;
61 ncols = length(columns);
62 axCell = cell(1, ncols); % We will also initialize the container of the graphical objects
63 
64 % 4. We will now iterate through the columns to find the rows inside each
65 % column. And inside the rows we will find the panels that contain the
66 % axes, and we will store those in the preinitialized cells
67 maxrows = 0;
68 for ii = ncols:-1:1
69  rows = columns(ii).Children;
70  nrows = length(rows);
71  maxrows = max(maxrows, nrows);
72  axCell{ncols-ii+1} = gobjects(nrows, 1);
73  for jj = nrows:-1:1 % Rows are also in reverse order (from bottom to top)
74  ax = rows(jj).Children(arrayfun(@(x) isa(x, 'matlab.graphics.axis.Axes'),rows(jj).Children));
75  axCell{ncols-ii+1}(nrows-jj+1) = ax;
76  end
77 end
78 
79 % 5. Exporting begins here. Create the figure, and define the subplot
80 % structure using the information inside 'axCell'
81 % Check for weird structures
82 for kk = 1:ncols
83  if mod(length(axCell{kk}),maxrows)
84  maxrows = maxrows * length(axCell{kk});
85  end
86 end
87 
88 hfig = figure();
89 for ii = 1:ncols
90  axrows = axCell{ii};
91  nax = length(axrows);
92  step = maxrows/nax;
93  for jj = 1:nax
94  a = copyobj(axCell{ii}(jj), hfig);
95  init = ii + (jj-1)*step*ncols;
96  fin = init + (step-1)*ncols;
97  aa = subplot(maxrows, ncols, init:ncols:fin, changeAxes(a));
98  changeLegend(aa);
99  end
100 end
101 
102 
103 m = uimenu(hfig,'Text','kVIS3');
104 uimenu(m, 'Text', 'Print Figure', 'Callback', @printcallback)
105 end

107 function a = changeAxes(a)
108 a.XLabel.Color = [0 0 0];
109 a.YLabel.Color = [0 0 0];
110 a.XColor = [0 0 0];
111 a.YColor = [0 0 0];

114 end
