88 %% PR 115-1897 quick fix
89 % Something in the existing plot will not load correctly when saved as a
90 % .fig file.
91 % in order to work around this, the subplots will be copied from the
92 % existing plot to a new plot before saving.
93

95 gcf;
96 f_new.Name = fs.Name;
97 w = length(f3.Children.Children);
98 h_max = 0;
99 for c=1:w
100  if length(f3.Children.Children(c,1).Children) > h_max
101  h_max = length(f3.Children.Children(c,1).Children);
102  end
103 end
104 
105 %%
106 % The current figure subplot positions increase in rows from bottom to top
107 % and columns from right to left. Therefore, we must iterate through them
108 % in reverse order to maintain the same positions in the new plot.
109 
110 n_spp = 1;
111 spp = 1;
112 for r = h_max:-1:1
113  for c = w:-1:1
114  if length(f3.Children.Children(c,1).Children) < r
115  continue
116  end
117  if isa(f3.Children.Children(c,1).Children(r,1).Children, 'matlab.graphics.axis.Axes')
118  if length(f3.Children.Children(c,1).Children(r,1).Children.Children) > 0
119  % ensure we dont copy an empty axes
120  sp = copyobj(f3.Children.Children(c,1).Children(r,1).Children,...
121  f_new);
122  sp_ax(n_spp, 1) = sp;
123  subplot(h_max, w, spp, sp_ax(n_spp, 1));
124  n_spp = n_spp + 1;
125  end
126  elseif length(f3.Children.Children(c,1).Children(r,1).Children) == 2
127  sp = copyobj([f3.Children.Children(c,1).Children(r,1).Children(2,1),...
128  f3.Children.Children(c,1).Children(r,1).Children(1,1)], f_new);
129  sp(2).Orientation = 'vertical';
130  sp(2).Location = 'westoutside';
131  sp_ax(n_spp, 1) = sp(1);
132  subplot(h_max, w, spp, sp_ax(n_spp, 1));
133  n_spp = n_spp + 1;
134  end

136  end
137 end
