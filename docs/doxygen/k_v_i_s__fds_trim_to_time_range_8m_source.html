21 function [fds, msg] = kVIS_fdsTrimToTimeRange(fds, TimeRange)
22 %
23 % Trim a fds structure to given time limits
24 %
25 
26 for I = 1 : size(fds.fdata, 2)
27 
28  if isempty(fds.fdata{fds.fdataRows.data, I})
29  continue;
30  end
31 
32  %
33  % relate time to data samples
34  %
35  t = fds.fdata{fds.fdataRows.data, I}(:,1);
36 
37  % start point - use first available sample if data is shorter
38  in = find(t <= TimeRange(1), 1, 'last' );
39 
40  if isempty(in)
41  in = 1;
42  end
43 
44  % end point - use last available sample if data is shorter
45  out= find(t >= TimeRange(2), 1, 'first');
46 
47  if isempty(out)
48  out = length(t);
49  end
50 
51  %
52  % update data length
53  %
54  data = fds.fdata{fds.fdataRows.data, I};
55 
56  fds.fdata{fds.fdataRows.data, I} = data(in:out,:);
57 end
58 
59 % edit event list to remove events outside new time range
60 eList = fds.eventList;
61 
62 EventID = [];
63 for i = 1:length(eList)
64 
65  if eList(i).start < TimeRange(1) || eList(i).start > TimeRange(2)
66  EventID = [EventID; i];
67  end
68 
69 end
70 
71 if ~isempty(EventID)
72  fds.eventList = kVIS_trimEventList(EventID, eList);
73  msg='Dataset trimming complete. Event list was adjusted.';
74 else
75  msg='Dataset trimming complete';
76 end
77 
78 %
79 % Update attributes
80 %
81 fds = kVIS_fdsUpdateAttributes(fds);
