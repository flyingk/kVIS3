21 function varargout = kVIS_createMap_Callback(hObject, ~, pathColorChannel)
22 % Function for plotting the flight path using plot_google_map
23 %
24 % matt & kai
25 
26 fds = kVIS_getCurrentFds(hObject);
27 
28 if ~isstruct(fds)
29  errordlg('Nothing loaded...')
30  return
31 end
32 
33 
34 if isempty(getpref('kVIS_prefs','google_maps_api_key'))
35  kVIS_terminalMsg('Enter Google Maps Static API key in preferences for satellite plot. Fallback to Open street map');
36  osm = true;
37 else
38  kVIS_terminalMsg('Generating satellite map plot...');
39  osm = false;
40 end
41 
42 % Get the track co-ords - assumed on a common time vector
43 [lat, lon, alt, t] = BSP_mapCoordsFcn(fds);
44 
45 % get selected time limits
46 xlim = kVIS_getDataRange(hObject, 'XLim');
47 % find corresponding samples
48 locs = find(t>xlim(1) & t<xlim(2));
49 
50 % create time vector for map plot @ 10 Hz
51 tNew = t(locs(1)):1/10:t(locs(end));
52 
53 lon = kVIS_reSample(lon, t, tNew);
54 lat = kVIS_reSample(lat, t, tNew);
55 alt = kVIS_reSample(alt, t, tNew);
56 
57 % track colouring
58 try
59  if isempty(pathColorChannel)
60  % use currently selected channel
61  [c, signalMeta] = kVIS_fdsGetCurrentChannel(hObject);
62  else
63  % use specified channel
64  pcc = strsplit(pathColorChannel,'/');
65  [c, signalMeta] = kVIS_fdsGetChannel(fds, pcc{1}, pcc{2});
66  end
67 
68  chan_name = [signalMeta.name ' [' signalMeta.unit ']'];
69 
70  c = kVIS_reSample(c, t, tNew);
71 catch
72  c = zeros(size(lon));
73  chan_name = [];
74 end
75 
76 
77 %
78 % create plot
79 %
80 if nargout == 0
81  targetPanel = kVIS_dataViewerGetActivePanel();
82  axes_handle = targetPanel.axesHandle;
83 else
84  figHdl = figure('Color','w','Visible','off');
85  axes_handle = axes(figHdl);
86 end
87 
88 cla(axes_handle, 'reset');
89 hold on
90 
91 h = scatter3(axes_handle, lon, lat, alt, 4, c);
92 h.Tag = 'MapPlot';
93 h.UserData.timeVec = tNew;
94 
95 
96 if (xlim(1)-0.2 <= min(t))
97  % Plot take off marker
98  plot(axes_handle, lon(1),lat(1),'wo','lineWidth',2,'markerSize',10);
99 else
100  % Plot in marker
101  text(axes_handle, lon(1),lat(1),'IN','color','w','HorizontalAlignment','center','FontWeight','bold');
102 end
103 
104 if (xlim(2)+0.2 >= max(t))
105  % Plot land marker
106  plot(axes_handle, lon(end),lat(end),'wx','lineWidth',2,'markerSize',10);
107 else
108  % Plot out marker
109  text(axes_handle, lon(end),lat(end),'OUT','color','w','HorizontalAlignment','center','FontWeight','bold');
110 end
111 
112 if osm == true
113  % Use OpenStreetMap which not requires a Google API key
114  % Add OpenStreetMap underlay
115  plot_openstreetmap('Alpha', 1, 'Scale', 1);
116 else
117  % Add google map underlay
118  plot_google_map( ...
119  'mapType','satellite', ...
120  'MapScale',2, ...
121  'Refresh',1, ...
122  'Axis', axes_handle,...
123  'apikey', getpref('kVIS_prefs','google_maps_api_key')...
124  );
125 end
126 
127 
128 % Pretty run
129 lbl = kVIS_generateLabels(signalMeta, []);
130 
131 view(axes_handle, 2)
132 
133 colormap(axes_handle, 'jet')
134 cb = colorbar(axes_handle,'Location','west');
135 cb.Color = 'w';
136 
137 if nargout == 0
138 
139  title(h.Parent,['Track color: ' lbl], 'Color','w', 'FontSize', 14, 'Interpreter', 'latex')
140  axes_handle.XColor = 'w';
141  axes_handle.YColor = 'w';
142  axes_handle.ZColor = 'w';
143  kVIS_axesResizeToContainer(axes_handle);
144  axes_handle.UIContextMenu = kVIS_createMapContextMenu(axes_handle);
145 else
146 
147  title(h.Parent,['Track color: ' lbl], 'Color','k', 'FontSize', 14, 'Interpreter', 'latex')
148  figHdl.Name = 'Flight Track';
149  varargout{1} = figHdl;
150 end
151 end
