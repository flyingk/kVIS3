21 function [ fds ] = kVIS_fdsUpgradeLegacy(fds)
22  % Upgrade the given FDS struct in legacy format (no formal specification)
23  % to a format understood by fds_upgrade_V_0_V_1_0.
24  % For KSID internal use only. Use fds_upgrade(fds) instead.
25 
26  assert(isstruct(fds));
27  assert(isfield(fds, 'fdata'));
28  assert(iscell(fds.fdata));
29 
30  flag = struct();
31  flag.UAVmainframe = false;
32 
33  if size(fds.fdata, 1) < 4
34 
35  % old file must be UAVmainframe...
36  flag.UAVmainframe = true;
37 
38  fdata(1,:) = fds.fdata(1,:);
39  fdata(2,:) = fds.varlab;
40  fdata(3,:) = fds.varunits;
41  fdata(4,:) = fds.fdata(2,:);
42  fds.fdata = fdata;
43 
44  fdataRows.Label = 1;
45  fdataRows.Vars = 2;
46  fdataRows.Units = 3;
47  fdataRows.Data = 4;
48  fds.fdataRows = fdataRows;
49 
50  fds.BoardSupportPackage = 'UAVmainframe';
51 
52  end
53 
54  assert(isfield(fds, 'fdataRows'));
55  assert(isstruct(fds.fdataRows));
56  assert(numel(fieldnames(fds.fdataRows)) == size(fds.fdata, 1));
57  assert(isfield(fds.fdataRows, 'Label'));
58  assert(isfield(fds.fdataRows, 'Vars'));
59  assert(isfield(fds.fdataRows, 'Units'));
60  assert(isfield(fds.fdataRows, 'Data'));
61 
62  if ~isfield(fds, 't')
63  fds.t = fds_get_channel(fds, 'Base', 'Time') - fds_get_channel(fds, 'Base', 'Time', 1);
64  end
65 
66  if ~isfield(fds, 'XLimit')
67  fds.XLimit = [fds.t(1) fds.t(end)];
68  end
69  if ~isfield(fds, 'YLimit')
70  fds.YLimit = [-100 100];
71  end
72 
73  % Add time synchronization info
74  if ~isfield(fds, 'offset')
75  fds.offset = 0;
76  end
77 
78  % Generate event list
79  if flag.UAVmainframe
80  if isfield(fds, 'config')
81  fds.ACconfig = fds.config;
82  else
83  fds.ACconfig = [];
84  end
85  fds = UAVmainframe_event_list(fds);
86  end
87 
88  % Upgrade event list
89  if ~isfield(fds, 'event_list') || isempty(fds.event_list)
90  fds.event_list = kVIS_fdsCreateEmptyEventList();
91  elseif iscell(fds.event_list)
92  EventList = kVIS_fdsCreateEmptyEventList();
93  for I = 1 : size(fds.event_list, 1)
94  K = numel(EventList) + 1;
95  EventList(K).Type = fds.event_list{I, 1};
96  EventList(K).Start = fds.event_list{I, 2};
97  EventList(K).End = fds.event_list{I, 3};
98  EventList(K).Plot = fds.event_list{I, 4};
99  end
100  fds.event_list = EventList;
101  end
102 
103  % Derive tree structure from data and labels
104  if ~isfield(fds.fdataRows, 'Parent')
105  % Derive structure from data and labels
106  fds.fdata = [ fds.fdata; cell(1, size(fds.fdata, 2)) ];
107  fds.fdataRows.Parent = size(fds.fdata, 1);
108  root_node = 0;
109  parent_node = root_node;
110  for idx = 1 : size(fds.fdata, 2)
111  node_name = fds.fdata{fds.fdataRows.Label, idx};
112  node_is_category = isempty(fds.fdata{fds.fdataRows.Vars, idx});
113  if node_is_category
114  node_name = strip(strip(node_name, '='));
115  fds.fdata{fds.fdataRows.Label, idx} = node_name;
116  parent_node = root_node;
117  end
118  fds.fdata{fds.fdataRows.Parent, idx} = parent_node;
119  if node_is_category
120  parent_node = idx;
121  end
122  end
123  end
124 
125  % Add TeX var labels field
126  if ~isfield(fds.fdataRows, 'VarLabels_TeX')
127  fds.fdata = [ fds.fdata; fds.fdata(fds.fdataRows.Vars, :) ];
128  fds.fdataRows.VarLabels_TeX = size(fds.fdata, 1);
129  end
130 
131  % Add tree state information
132  if ~isfield(fds.fdataRows, 'GUIexpanded')
133  fds.fdata = [ fds.fdata; cell(1, size(fds.fdata, 2)) ];
134  fds.fdataRows.GUIexpanded = size(fds.fdata, 1);
135  fds.fdata(fds.fdataRows.GUIexpanded, :) = {false};
136  end
137  if ~isfield(fds.fdataRows, 'GUIselected')
138  fds.fdata = [ fds.fdata; cell(1, size(fds.fdata, 2)) ];
139  fds.fdataRows.GUIselected = size(fds.fdata, 1);
140  fds.fdata(fds.fdataRows.GUIselected, :) = {false};
141  end
142 
143  % Determine array sizes
144  fds.npoints = length(fds.t);
145  fds.nfiles = size(fds.fdata,2);
146  fds.nchnls = cellfun(@(x) size(x, 2), fds.fdata(fds.fdataRows.Data, :));
147 
148  % Update sample rate
149  fds.SampleRate = 1/((fds.t(end) - fds.t(1))/length(fds.t));
150 
151  % Add BoardSupportPackage field
152  if ~isfield(fds, 'BoardSupportPackage')
153  fds.BoardSupportPackage = '';
154  end
155 
156  fds.fdsVersion = 0;
157 
158 end
