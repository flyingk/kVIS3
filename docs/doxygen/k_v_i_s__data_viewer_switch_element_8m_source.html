30 function kVIS_dataViewerSwitchElement(selectedPanel, ~)
31 % scenarios
32 % - no previous selection: make current panel active
33 % - previous selection exists:
34 % - no link request: make current panel active, previous panel inactive
35 % - link request: link panels, make target active
36 
37 %
38 % if the panel is selcted by the axis button down fcn, get the Parent panel
39 % handle
40 %
41 if isprop(selectedPanel, 'XLim')
42  selectedPanel = selectedPanel.Parent;
43 end
44 
45 handles = guidata(selectedPanel);
46 
47 %
48 % get previously selected panel
49 %
50 previousPanel = kVIS_dataViewerGetActivePanel;
51 
52 if isempty(previousPanel)
53  %
54  % first plot or nothing selected after delete
55  %
56  selectedPanel.HighlightColor = 'c';
57 
58 elseif previousPanel == selectedPanel
59  %
60  % callback goes all the time...
61  %
62  return
63 
64 elseif strcmp(selectedPanel.Tag, 'fftplot') || strcmp(selectedPanel.Tag, 'corrplot')
65  %
66  % don't select special plots
67  %
68  return
69 
70 else
71  %
72  % link request?
73  %
74  if previousPanel.linkPending == true
75  % link panels
76  previousPanel.linkFrom = selectedPanel;
77  selectedPanel.linkTo = previousPanel;
78  previousPanel.linkPending = false;
79 
80  % indicate link
81  rn = 1;%rand;
82  selectedPanel.BackgroundColor = getpref('kVIS_prefs','uiBackgroundColour') + 0.2*rn;
83  previousPanel.BackgroundColor = getpref('kVIS_prefs','uiBackgroundColour') + 0.2*rn;
84 
85  previousPanel.HighlightColor = getpref('kVIS_prefs','uiBackgroundColour');
86  selectedPanel.HighlightColor = 'c';
87 
88  % install listener on linked timeplot
89  if strcmp(previousPanel.Tag, 'fftplot')
90  % save listener handle in link target
91  previousPanel.plotChangedListener = addlistener(selectedPanel,...
92  'plotChanged','PostSet',@kVIS_fftUpdate);
93 
94  kVIS_fftUpdate([],[])
95 
96  elseif strcmp(previousPanel.Tag, 'corrplot')
97 
98  previousPanel.plotChangedListener = addlistener(selectedPanel,...
99  'plotChanged','PostSet',@kVIS_correlationPlot);
100 
101  kVIS_correlationPlot([],[])
102 
103  end
104 
105  else
106  %
107  % save limits
108  %
109  previousPanel.xLim = kVIS_getDataRange(selectedPanel, 'XLim');
110  previousPanel.yLim = kVIS_getDataRange(selectedPanel, 'YLim');
111  %
112  % set selected panel active, previous panel inactive
113  %
114  selectedPanel.HighlightColor = 'c';
115 
116  previousPanel.HighlightColor = getpref('kVIS_prefs','uiBackgroundColour');
117 
118 % kVIS_setDataRange(selectedPanel, 'XLim', selectedPanel.xLim);
119  kVIS_setDataRange(selectedPanel, 'YLim', selectedPanel.yLim);
120  end

122 end
